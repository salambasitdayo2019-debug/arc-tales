<script>
    // Fake contract for demo (replace with real contract later)
    let wallet = null;
    let stories = [];

    function connectWallet() {
        // Check if MetaMask exists
        if (typeof window.ethereum === 'undefined') {
            alert('Please install MetaMask: metamask.io');
            return;
        }
        
        // Connect
        window.ethereum.request({ method: 'eth_requestAccounts' })
            .then(accounts => {
                wallet = accounts[0];
                document.getElementById('walletBtn').textContent = 
                    wallet.slice(0,6) + '...' + wallet.slice(-4);
                loadStories();
            })
            .catch(err => alert('Connection failed: ' + err.message));
    }

    function togglePrice() {
        const checkbox = document.getElementById('isPremium');
        const priceBox = document.getElementById('priceBox');
        priceBox.className = checkbox.checked ? '' : 'hidden';
    }

    async function uploadStory() {
        if (!wallet) {
            alert('Connect wallet first!');
            return;
        }

        const title = document.getElementById('title').value;
        const content = document.getElementById('content').value;
        const genre = document.getElementById('genre').value;
        const isPremium = document.getElementById('isPremium').checked;
        const price = document.getElementById('unlockPrice').value;

        if (!title || !content || !genre) {
            alert('Fill all fields!');
            return;
        }

        const btn = document.getElementById('uploadBtn');
        btn.textContent = 'Processing...';
        btn.disabled = true;

        try {
            // STEP 1: Upload to free IPFS (web3.storage)
            const ipfsHash = await uploadToIPFS(content);
            
            // STEP 2: Pay 1 USDC (this would be real blockchain transaction)
            // For demo, we simulate it
            await payUSDC(1);
            
            // STEP 3: Save story
            const story = {
                id: stories.length,
                title,
                genre,
                author: wallet,
                contentHash: ipfsHash,
                preview: content.slice(0, 100) + '...',
                isPremium,
                unlockPrice: isPremium ? price : 0,
                timestamp: new Date().toLocaleString()
            };
            
            stories.unshift(story);
            
            alert('âœ… Story published! Paid 1 USDC.');
            clearForm();
            loadStories();
            
        } catch (err) {
            alert('Error: ' + err.message);
        }
        
        btn.textContent = 'Publish Story (Pay 1 USDC)';
        btn.disabled = false;
    }

    async function uploadToIPFS(content) {
        // FREE: Using a simple workaround
        // In production, use web3.storage (free 5GB)
        
        // For demo: Create data URL (not real IPFS but works for testing)
        const blob = new Blob([content], {type: 'text/plain'});
        return 'ipfs-demo-' + Math.random().toString(36).slice(2);
    }

    async function payUSDC(amount) {
        // This would be real USDC transfer on Arc Testnet
        // For demo, we just wait 1 second
        return new Promise(resolve => setTimeout(resolve, 1000));
    }

    function loadStories() {
        const list = document.getElementById('storiesList');
        
        if (stories.length === 0) {
            list.innerHTML = '<p>No stories yet. Be the first!</p>';
            return;
        }
        
        list.innerHTML = stories.map(s => `
            <div class="story-preview">
                <h3>${s.title}</h3>
                <span class="badge">${s.genre}</span>
                <span class="badge">${s.isPremium ? 'ðŸ’° ' + s.unlockPrice + ' USDC' : 'FREE'}</span>
                <p style="color: #666; margin: 10px 0;">${s.preview}</p>
                <small>By ${s.author.slice(0,6)}...${s.author.slice(-4)} â€¢ ${s.timestamp}</small>
            </div>
        `).join('');
    }

    function clearForm() {
        document.getElementById('title').value = '';
        document.getElementById('content').value = '';
        document.getElementById('genre').value = '';
        document.getElementById('isPremium').checked = false;
        togglePrice();
    }
</script>
